\subsection{VGG Block Pattern}

\FormulaName{VGG-style block}
\[
A^{[l]}
=
\mathrm{Pool}
\left(
\mathrm{Conv}_{3\times3}^{(k)}
(A^{[l-1]})
\right).
\]
\Use{Repeated small-filter convolution block.}

\subsection{Residual Block}

\FormulaName{Identity shortcut}
\[
A^{[l+2]}
=
g
\left(
Z^{[l+2]} + A^{[l]}
\right).
\]
\FormulaName{Projection shortcut}
\[
A^{[l+2]}
=
g
\left(
Z^{[l+2]} + W_s A^{[l]}
\right).
\]
\Use{Residual connection.}
\Notes{For identity shortcuts, $Z^{[l+2]}$ and $A^{[l]}$ must be shape-compatible. For projection shortcuts, $W_s A^{[l]}$ matches the shape of $Z^{[l+2]}$.}

\subsection{Inception Concatenation}

\FormulaName{Channel concatenation}
\[
A^{[l]}
=
\operatorname{Concat}_{\mathrm{channels}}
\left(
A_{1\times1},
A_{3\times3},
A_{5\times5},
A_{\mathrm{pool}}
\right).
\]
\[
n_C^{[l]}
=
n_{C,1\times1}
+
n_{C,3\times3}
+
n_{C,5\times5}
+
n_{C,\mathrm{pool}}.
\]
\Use{Parallel convolution branches with channel concatenation.}

\subsection{1x1 Bottleneck Parameter Reduction}

\FormulaName{Without bottleneck}
\[
\#\mathrm{params}
=
(f^2 n_C^{\mathrm{in}} + 1)n_C^{\mathrm{out}}.
\]
\FormulaName{With 1x1 bottleneck width $r$}
\[
\#\mathrm{params}
=
(1 \cdot 1 \cdot n_C^{\mathrm{in}} + 1)r
+
(f^2 r + 1)n_C^{\mathrm{out}}.
\]
\Use{Parameter reduction before expensive convolution.}
