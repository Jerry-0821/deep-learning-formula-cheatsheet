\section{Face Recognition and Neural Style Transfer}

\subsection{Embedding Function}

\FormulaName{Image embedding}
\[
f(x) \in \R^d.
\]
\Use{Map input image to embedding vector.}

\subsection{Pairwise Distance}

\FormulaName{Embedding-space distance}
\[
d(x_i,x_j)
=
\left\lVert f(x_i)-f(x_j) \right\rVert_2^2.
\]
\Use{Embedding-space distance.}

\subsection{Siamese / Triplet Loss}

\FormulaName{Triplet}
\[
(A,P,N).
\]
\Use{Anchor, positive, negative.}

\FormulaName{Triplet loss}
\[
\loss_{\mathrm{triplet}}
=
\max
\left(
\left\lVert f(A)-f(P) \right\rVert_2^2
-
\left\lVert f(A)-f(N) \right\rVert_2^2
+
\alpha,
0
\right).
\]
\Use{Separate positive and negative pairs by margin $\alpha$.}

\subsection{Content Cost}

\FormulaName{Content objective}
\[
\cost_{\mathrm{content}}(C,G)
=
\frac{1}{2}
\left\lVert
A_C^{[l]} - A_G^{[l]}
\right\rVert_F^2.
\]
\Use{Content similarity between generated and content image activations.}

\subsection{Gram Matrix}

\FormulaName{Style Gram matrix}
\[
F^{[l]} \in \R^{n_C^{[l]} \times (n_H^{[l]}n_W^{[l]})},
\qquad
G^{[l]}
=
F^{[l]}(F^{[l]})\transpose.
\]
\[
G^{[l]} \in \R^{n_C^{[l]} \times n_C^{[l]}}.
\]
\Use{Channel-correlation matrix for style.}

\subsection{Style Cost}

\FormulaName{Layer style cost}
\[
\cost_{\mathrm{style}}^{[l]}(S,G)
=
\frac{1}{
(2 n_H^{[l]} n_W^{[l]} n_C^{[l]})^2
}
\left\lVert
G_S^{[l]} - G_G^{[l]}
\right\rVert_F^2.
\]
\FormulaName{Total style cost}
\[
\cost_{\mathrm{style}}(S,G)
=
\sum_l
\lambda_l
\cost_{\mathrm{style}}^{[l]}(S,G).
\]
\Use{Style similarity across selected layers.}

\subsection{Total Neural Style Transfer Objective}

\FormulaName{Style transfer objective}
\[
\cost(G)
=
\alpha \cost_{\mathrm{content}}(C,G)
+
\beta \cost_{\mathrm{style}}(S,G).
\]
\Use{Neural style transfer objective.}
\Notes{Here $\alpha,\beta$ are style-transfer weights, not optimizer learning rate or BatchNorm shift.}
