\section{Object Detection and YOLO}

\subsection{Bounding Box Representation}

\FormulaName{Box parameterizations}
\[
b=(b_x,b_y,b_h,b_w),
\qquad
b=(x_{\min},y_{\min},x_{\max},y_{\max}).
\]
\Use{Bounding box coordinates.}
\Notes{Use one representation consistently within a formula.}

\subsection{Intersection over Union}

\FormulaName{IoU}
\[
\operatorname{IoU}(B_1,B_2)
=
\frac{
\operatorname{area}(B_1 \cap B_2)
}{
\operatorname{area}(B_1 \cup B_2)
}.
\]
\Use{Bounding-box overlap score.}

\subsection{YOLO Grid Output Shape}

\FormulaName{Course-style YOLO output}
\[
S \times S \times (B \cdot 5 + C).
\]
\[
\text{Batch output}
\in
\R^{m \times S \times S \times (B \cdot 5 + C)}.
\]
\[
(p_c,b_x,b_y,b_h,b_w).
\]
\Use{YOLO grid-cell prediction tensor.}
\Notes{Modern YOLO variants may use different head designs; deferred to V2.}

\subsection{Objectness and Class Scores}

\FormulaName{Class-specific confidence}
\[
p_c=P(\mathrm{object}),
\qquad
P(c_j \mid \mathrm{object}).
\]
\[
\mathrm{score}_j
=
p_c\,P(c_j \mid \mathrm{object}).
\]
\Use{Class confidence score for detection.}

\subsection{Anchor Boxes}

\FormulaName{Anchor assignment}
\[
\text{anchor}
=
\argmax_{a}
\operatorname{IoU}(B_{\mathrm{true}}, A_a).
\]
\Use{Assign ground-truth box to best matching anchor.}

\subsection{Non-max Suppression}

\FormulaName{Selection and suppression}
\[
B^*
=
\argmax_{B_i}
\mathrm{score}(B_i).
\]
\[
\operatorname{IoU}(B_i,B^*) > \tau
\quad
\Rightarrow
\quad
\text{discard } B_i.
\]
\Use{Remove duplicate boxes.}
