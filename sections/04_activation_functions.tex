\subsection{Derivative Table}

\FormulaName{Softmax Jacobian}
\[
\frac{\partial a_i}{\partial z_j}
=
a_i(\delta_{ij}-a_j).
\]
\Shapes{$z,a \in \R^{K \times 1}$ for one example.}
\Use{Softmax derivative reference.}

\begin{center}
\begin{tabular}{lll}
\toprule
Symbol & Meaning & Shape / domain \\
\midrule
$z$ & scalar pre-activation or logit vector & $\R$ or $\R^{K \times 1}$ \\
$a$ & activation output & same leading shape as $z$ \\
$K$ & number of classes & $\mathbb{N}$ \\
$\delta_{ij}$ & Kronecker delta & $\{0,1\}$ \\
$\lambda_{\mathrm{leak}}$ & leaky ReLU slope & $0 < \lambda_{\mathrm{leak}} < 1$ \\
\bottomrule
\end{tabular}
\end{center}
